Folding and visualizing the baseband data¶

As the baseband data is differnet from filterband data, we have to fold the baseband data and visualize them into different phases.

In [30]:
#integrate 5 samples
step = 2**8 # divide into 2**8 spaces
int_size = 16 # folding size
n_spectra = int(len(bbsig1.data[0])/step/int_size) # size:(n_spectra, step)
spec_out = np.zeros((n_spectra,step), dtype=complex)
for i in range(n_spectra):
    for j in range(int_size):
        #print(i,j)
        spec = np.fft.fft(bbsig1.data[0][(int_size*i+j)*step:(int_size*i+j+1)*step])
        spec_out[i] += spec*spec.conjugate()
